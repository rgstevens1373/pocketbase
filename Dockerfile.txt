# Build stage
FROM golang:1.22-alpine AS builder

RUN apk add --no-cache curl unzip

# Download PocketBase binary
RUN curl -L -o /tmp/pb.zip https://github.com/pocketbase/pocketbase/releases/download/v0.21.1/pocketbase_0.21.1_linux_amd64.zip \
    && unzip /tmp/pb.zip -d /tmp

# Final stage
FROM alpine:3.18

WORKDIR /app

# Copy PocketBase binary
COPY --from=builder /tmp/pocketbase ./pocketbase

# Copy your start script
COPY start.sh .

# Ensure PocketBase is executable
RUN chmod +x pocketbase start.sh

EXPOSE 8090

CMD ["./start.sh"]
